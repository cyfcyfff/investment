const CURRENCY_SYMBOLS: Record<string, string> = {
  USD: '$',
  EUR: '€',
  GBP: '£',
  CNY: '¥',
  JPY: '¥',
}

export function formatCurrency(value: number, currency: string = 'USD', decimals: number = 2): string {
  const symbol = CURRENCY_SYMBOLS[currency] ?? currency
  const formatted = Math.abs(value).toLocaleString('en-US', {
    minimumFractionDigits: decimals,
    maximumFractionDigits: decimals,
  })
  return value < 0 ? `-${symbol}${formatted}` : `${symbol}${formatted}`
}

export function formatPercent(value: number, decimals: number = 2): string {
  return `${(value * 100).toFixed(decimals)}%`
}

export function formatNumber(value: number): string {
  return value.toLocaleString('en-US')
}

export function generateId(): string {
  return `${Date.now()}-${Math.random().toString(36).substring(2, 9)}`
}

const VALID_CURRENCIES = new Set([
  'USD', 'EUR', 'GBP', 'JPY', 'CNY', 'AUD', 'CAD', 'CHF', 'HKD',
  'SGD', 'KRW', 'INR', 'TWD', 'BRL', 'MXN', 'ZAR', 'SEK', 'NOK',
  'DKK', 'NZD', 'THB', 'MYR', 'PHP', 'IDR', 'VND', 'RUB', 'TRY',
])

export function isValidTicker(ticker: string): boolean {
  return /^[A-Z0-9][A-Z0-9.\-]*[A-Z0-9]$|^[A-Z0-9]$/.test(ticker) && ticker.length >= 1
}

export function isValidCurrency(code: string): boolean {
  return VALID_CURRENCIES.has(code)
}

export function isValidDate(dateStr: string): boolean {
  const date = new Date(dateStr)
  return !isNaN(date.getTime()) && dateStr.match(/^\d{4}-\d{2}-\d{2}$/) !== null
}

export function isValidWeight(w: number): boolean {
  return typeof w === 'number' && !isNaN(w) && w >= 0 && w <= 1
}

import { Category, CATEGORIES } from '../types'
import type { AssetHolding, RebalanceConfig } from '../types'

export function calcMarketValue(quantity: number, price: number, fxRate: number): number {
  return quantity * price * fxRate
}

export function calculateCategoryWeights(
  holdings: AssetHolding[],
  prices: Record<string, number>,
  fxRates: Record<string, number>,
  baseCurrency: string,
): Record<Category, number> {
  if (holdings.length === 0) {
    return {
      [Category.STOCKS]: 0,
      [Category.LONG_BONDS]: 0,
      [Category.GOLD]: 0,
      [Category.CASH]: 0,
    }
  }

  const categoryValues: Record<Category, number> = {
    [Category.STOCKS]: 0,
    [Category.LONG_BONDS]: 0,
    [Category.GOLD]: 0,
    [Category.CASH]: 0,
  }

  for (const h of holdings) {
    const price = prices[h.ticker]
    if (price === undefined) continue

    const fx = h.currency === baseCurrency ? 1 : (fxRates[h.currency] ?? 1)
    const value = calcMarketValue(h.quantity, price, fx)
    categoryValues[h.category] += value
  }

  const totalValue = CATEGORIES.reduce((sum, cat) => sum + categoryValues[cat], 0)
  if (totalValue === 0) {
    return {
      [Category.STOCKS]: 0,
      [Category.LONG_BONDS]: 0,
      [Category.GOLD]: 0,
      [Category.CASH]: 0,
    }
  }

  const weights: Record<Category, number> = {
    [Category.STOCKS]: 0,
    [Category.LONG_BONDS]: 0,
    [Category.GOLD]: 0,
    [Category.CASH]: 0,
  }

  for (const cat of CATEGORIES) {
    weights[cat] = categoryValues[cat] / totalValue
  }

  return weights
}

export interface RebalanceCheckResult {
  triggered: boolean
  reason: string
  breachCategories: Category[]
}

export function isRebalanceTriggered(
  weights: Record<Category, number>,
  config: RebalanceConfig,
  lastRebalanceDate?: string,
  now?: string,
): RebalanceCheckResult {
  const breachCategories: Category[] = []

  for (const cat of CATEGORIES) {
    const w = weights[cat]
    if (w < config.bands.low || w > config.bands.high) {
      breachCategories.push(cat)
    }
  }

  if (breachCategories.length > 0) {
    return {
      triggered: true,
      reason: `带宽触发: ${breachCategories.map(c => c).join(', ')} 超出范围 [${config.bands.low}, ${config.bands.high}]`,
      breachCategories,
    }
  }

  // Check annual review
  if (lastRebalanceDate) {
    const last = new Date(lastRebalanceDate)
    const currentDate = now ? new Date(now) : new Date()
    const daysSince = (currentDate.getTime() - last.getTime()) / (1000 * 60 * 60 * 24)

    if (daysSince >= 365) {
      return {
        triggered: true,
        reason: `年度校准: 距上次再平衡已 ${Math.floor(daysSince)} 天`,
        breachCategories: [],
      }
    }
  }

  return {
    triggered: false,
    reason: '',
    breachCategories: [],
  }
}

import type { Quote } from '../types'

const YAHOO_QUOTE_URL = 'https://query1.finance.yahoo.com/v8/finance/chart/'

interface YahooMeta {
  regularMarketPrice: number
  previousClose: number
  currency: string
  regularMarketTime: number
}

interface YahooResult {
  meta: YahooMeta
}

interface YahooResponse {
  chart: {
    result: YahooResult[]
  }
}

function parseYahooQuote(ticker: string, meta: YahooMeta): Quote {
  return {
    ticker,
    price: meta.regularMarketPrice,
    currency: meta.currency,
    change: meta.regularMarketPrice - meta.previousClose,
    changePercent: meta.previousClose !== 0
      ? (meta.regularMarketPrice - meta.previousClose) / meta.previousClose
      : 0,
    asOf: new Date(meta.regularMarketTime * 1000).toISOString(),
    source: 'YAHOO',
  }
}

export async function fetchQuote(ticker: string): Promise<Quote> {
  const url = `${YAHOO_QUOTE_URL}${encodeURIComponent(ticker)}?interval=1d&range=1d`

  const response = await fetch(url)
  if (!response.ok) {
    throw new Error(`Failed to fetch quote for ${ticker}: ${response.status}`)
  }

  const data: YahooResponse = await response.json()
  const result = data.chart.result[0]

  return parseYahooQuote(ticker, result.meta)
}

export async function fetchQuotes(tickers: string[]): Promise<Quote[]> {
  if (tickers.length === 0) return []

  // Fetch in parallel (Yahoo supports batch via comma-separated)
  // For reliability, fetch individually in parallel
  const results = await Promise.allSettled(
    tickers.map(t => fetchQuote(t)),
  )

  const quotes: Quote[] = []
  for (let i = 0; i < results.length; i++) {
    const r = results[i]
    if (r.status === 'fulfilled') {
      quotes.push(r.value)
    } else {
      console.warn(`Failed to fetch quote for ${tickers[i]}:`, r.reason)
    }
  }

  return quotes
}

import type { FxRate } from '../types'

export async function fetchFxRate(base: string, quote: string): Promise<FxRate> {
  if (base === quote) {
    return {
      base,
      quote,
      rate: 1,
      asOf: new Date().toISOString(),
      source: 'SAME_CURRENCY',
    }
  }

  // Yahoo Finance uses pair format: BASEQUOTE=X (e.g., USDCNY=X)
  const pair = `${base}${quote}=X`
  const url = `https://query1.finance.yahoo.com/v8/finance/chart/${encodeURIComponent(pair)}?interval=1d&range=1d`

  const response = await fetch(url)
  if (!response.ok) {
    throw new Error(`Failed to fetch FX rate ${base}/${quote}: ${response.status}`)
  }

  const data = await response.json()
  const meta = data.chart.result[0].meta

  return {
    base,
    quote,
    rate: meta.regularMarketPrice,
    asOf: new Date(meta.regularMarketTime * 1000).toISOString(),
    source: 'YAHOO',
  }
}

export async function fetchFxRates(
  pairs: Array<{ base: string; quote: string }>,
): Promise<Record<string, number>> {
  const rates: Record<string, number> = {}
  const results = await Promise.allSettled(
    pairs.map(async ({ base, quote }) => {
      const fx = await fetchFxRate(base, quote)
      return { key: `${base}-${quote}`, rate: fx.rate }
    }),
  )

  for (const r of results) {
    if (r.status === 'fulfilled') {
      rates[r.value.key] = r.value.rate
    }
  }

  return rates
}

function loadFromStorage<T>(key: string, fallback: T): T {
  try {
    const raw = localStorage.getItem(key)
    return raw ? JSON.parse(raw) : fallback
  } catch {
    return fallback
  }
}

function saveToStorage<T>(key: string, value: T): void {
  localStorage.setItem(key, JSON.stringify(value))
}

interface ConfigState {
  rebalanceConfig: RebalanceConfig
  appConfig: AppConfig

  updateRebalanceConfig: (updates: Partial<RebalanceConfig>) => void
  updateAppConfig: (updates: Partial<AppConfig>) => void
  resetToDefaults: () => void

interface QuoteState {
  quotes: Record<string, Quote>
  fxRates: Record<string, FxRate>
  loading: boolean
  lastUpdated: string | null
  error: string | null

  refreshQuotes: (tickers: string[]) => Promise<void>
  refreshFxRates: (pairs: Array<{ base: string; quote: string }>) => Promise<void>
  refreshAll: (
    tickers: string[],
    currencies: string[],
    baseCurrency: string,
  ) => Promise<void>

  fxRates: {},

  lastUpdated: null,
  error: null,

    set({ loading: true, error: null })
    try {
      const results = await fetchQuotes(tickers)
      const quoteMap: Record<string, Quote> = {}
      for (const q of results) {
        quoteMap[q.ticker] = q
      }
      set(state => ({
        quotes: { ...state.quotes, ...quoteMap },
        lastUpdated: new Date().toISOString(),
        loading: false,
      }))
    } catch (e) {
      set({ error: String(e), loading: false })
    }
  },

  refreshFxRates: async (pairs) => {
    try {
      for (const { base, quote } of pairs) {
        if (base === quote) continue
        const rate = await fetchFxRate(base, quote)
        set(state => ({
          fxRates: { ...state.fxRates, [`${base}-${quote}`]: rate },
        }))
      }
    } catch (e) {
      console.warn('FX rate fetch failed:', e)
    }
  },

  refreshAll: async (tickers, currencies, baseCurrency) => {
    const uniqueCurrencies = [...new Set(currencies.filter(c => c !== baseCurrency))]
    const pairs = uniqueCurrencies.map(c => ({ base: c, quote: baseCurrency }))

    // Fetch quotes and FX rates in parallel
    await Promise.all([
      get().refreshQuotes(tickers),
      get().refreshFxRates(pairs),
    ])
  },
}))

import { useEffect, useMemo } from 'react'
import { Row, Col, Card, Statistic, Alert, Spin, Typography } from 'antd'
import { ReloadOutlined, WarningOutlined, CheckCircleOutlined } from '@ant-design/icons'
import { usePortfolioStore } from '../../stores/portfolioStore'
import { useQuoteStore } from '../../stores/quoteStore'
import { useConfigStore } from '../../stores/configStore'
import { calculateCategoryWeights, isRebalanceTriggered, calcMarketValue } from '../../services/calcService'
import { Category, CATEGORIES } from '../../types'
import { formatCurrency } from '../../utils/formatters'
import WeightPieChart from '../../components/charts/WeightPieChart'
import BandGauge from '../../components/charts/BandGauge'

const { Title } = Typography

export default function Dashboard() {
  const { holdings, loadHoldings } = usePortfolioStore()
  const { quotes, fxRates, loading, lastUpdated, refreshAll } = useQuoteStore()
  const { appConfig, rebalanceConfig } = useConfigStore()

  useEffect(() => {
    loadHoldings()
  }, [loadHoldings])

  useEffect(() => {
    if (holdings.length > 0) {
      const tickers = holdings.map(h => h.ticker)
      const currencies = holdings.map(h => h.currency)
      refreshAll(tickers, currencies, appConfig.baseCurrency)
    }
  }, [holdings, appConfig.baseCurrency, refreshAll])

  const prices = useMemo(() => {
    const p: Record<string, number> = {}
    for (const [ticker, quote] of Object.entries(quotes)) {
      p[ticker] = quote.price
    }
    return p
  }, [quotes])

  const fxMap = useMemo(() => {
    const m: Record<string, number> = {}
    for (const [key, fx] of Object.entries(fxRates)) {
      m[key] = fx.rate
    }
    return m
  }, [fxRates])

  const weights = useMemo(
    () => calculateCategoryWeights(holdings, prices, fxMap, appConfig.baseCurrency),
    [holdings, prices, fxMap, appConfig.baseCurrency],
  )

  const totalValue = useMemo(
    () => CATEGORIES.reduce((sum, cat) => sum + weights[cat] * holdings.reduce((s, h) => {
      const price = prices[h.ticker]
      if (price === undefined) return s
      const fx = h.currency === appConfig.baseCurrency ? 1 : (fxMap[`${h.currency}-${appConfig.baseCurrency}`] ?? 1)
      return s + h.quantity * price * fx
    }, 0) / Math.max(CATEGORIES.reduce((sum, cat) => sum + weights[cat], 0), 1), 0),
    [holdings, prices, fxMap, weights, appConfig.baseCurrency],
  )

  // Simpler total calculation
  const portfolioTotal = useMemo(() => {
    let total = 0
    for (const h of holdings) {
      const price = prices[h.ticker]
      if (price === undefined) continue
      const fx = h.currency === appConfig.baseCurrency ? 1 : (fxMap[`${h.currency}-${appConfig.baseCurrency}`] ?? 1)
      total += calcMarketValue(h.quantity, price, fx)
    }
    return total
  }, [holdings, prices, fxMap, appConfig.baseCurrency])

  const rebalanceCheck = useMemo(
    () => isRebalanceTriggered(weights, rebalanceConfig),
    [weights, rebalanceConfig],
  )

  if (holdings.length === 0) {
    return (
      <div>
        <Title level={3}>仪表盘</Title>
        <Alert
          message="欢迎使用 PermPort"
          description="请先前往「持仓管理」添加你的第一个持仓。"
          type="info"
          showIcon
        />
      </div>
    )
  }

  return (
    <div>
      <Title level={3}>仪表盘</Title>

      <Row gutter={[16, 16]} style={{ marginBottom: 24 }}>
        <Col xs={24} sm={12} lg={6}>
          <Card>
            <Statistic
              title="组合总市值"
              value={portfolioTotal}
              formatter={(v) => formatCurrency(Number(v), appConfig.baseCurrency)}
            />
          </Card>
        </Col>
        <Col xs={24} sm={12} lg={6}>
          <Card>
            <Statistic title="持仓数量" value={holdings.length} />
          </Card>
        </Col>
        <Col xs={24} sm={12} lg={6}>
          <Card>
            <Statistic
              title="数据更新时间"
              value={lastUpdated ? new Date(lastUpdated).toLocaleTimeString() : '-'}
            />
          </Card>
        </Col>
        <Col xs={24} sm={12} lg={6}>
          <Card>
            <Statistic
              title="再平衡状态"
              value={rebalanceCheck.triggered ? '需再平衡' : '正常'}
              valueStyle={{ color: rebalanceCheck.triggered ? '#ff4d4f' : '#52c41a' }}
              prefix={rebalanceCheck.triggered ? <WarningOutlined /> : <CheckCircleOutlined />}
            />
          </Card>
        </Col>
      </Row>

      {rebalanceCheck.triggered && (
        <Alert
          message={rebalanceCheck.reason}
          type="warning"
          showIcon
          style={{ marginBottom: 24 }}
        />
      )}

      <Row gutter={[16, 16]}>
        <Col xs={24} lg={12}>
          <Card title="资产配比">
            {loading ? <Spin /> : <WeightPieChart weights={weights} config={rebalanceConfig} />}
          </Card>
        </Col>

            {loading ? <Spin /> : <BandGauge weights={weights} config={rebalanceConfig} />}

    </div>
  )

**Step 1: Implement Settings page**

Replace `src/pages/Settings/index.tsx`:

```tsx
import { Card, Form, Select, InputNumber, Slider, Button, Divider, Typography, Alert, Space } from 'antd'
import { useConfigStore } from '../../stores/configStore'
import { Category, CATEGORY_LABELS, CATEGORIES, BAND_PRESETS, DEFAULT_REBALANCE_CONFIG, DEFAULT_APP_CONFIG } from '../../types'
import type { BandPreset } from '../../types'

const { Title, Text } = Typography

const BAND_PRESET_OPTIONS = [
  { value: 'CONSERVATIVE', label: '保守 (15%-35%)' },
  { value: 'MODERATE', label: '中等 (20%-30%)' },
  { value: 'AGGRESSIVE', label: '积极 (20%-30%)' },
  { value: 'CUSTOM', label: '自定义' },
]

const CURRENCY_OPTIONS = [
  { value: 'USD', label: 'USD - 美元' },
  { value: 'CNY', label: 'CNY - 人民币' },
  { value: 'EUR', label: 'EUR - 欧元' },
  { value: 'JPY', label: 'JPY - 日元' },
  { value: 'GBP', label: 'GBP - 英镑' },
  { value: 'HKD', label: 'HKD - 港币' },
]

export default function Settings() {
  const { rebalanceConfig, appConfig, updateRebalanceConfig, updateAppConfig, resetToDefaults } = useConfigStore()

  const totalWeight = CATEGORIES.reduce((s, c) => s + rebalanceConfig.targets[c], 0)
  const weightValid = Math.abs(totalWeight - 1) < 0.001

  return (
    <div>
      <Title level={3}>设置</Title>

      <Card title="基准货币" style={{ marginBottom: 16 }}>
        <Form layout="vertical">
          <Form.Item label="基准货币">
            <Select
              value={appConfig.baseCurrency}
              onChange={v => updateAppConfig({ baseCurrency: v })}
              options={CURRENCY_OPTIONS}
              style={{ width: 200 }}
            />
          </Form.Item>
          <Form.Item label="行情刷新间隔（分钟）">
            <InputNumber
              value={appConfig.quoteRefreshInterval}
              onChange={v => updateAppConfig({ quoteRefreshInterval: v ?? 15 })}
              min={1}
              max={60}
              style={{ width: 200 }}
            />
          </Form.Item>
        </Form>
      </Card>

      <Card title="目标权重" style={{ marginBottom: 16 }}>
        <Alert
          message={weightValid ? '权重总和为 100%' : `当前总和: ${(totalWeight * 100).toFixed(1)}%，请调整为 100%`}
          type={weightValid ? 'success' : 'warning'}
          showIcon
          style={{ marginBottom: 16 }}
        />
        <Form layout="vertical">
          {CATEGORIES.map(cat => (
            <Form.Item key={cat} label={`${CATEGORY_LABELS[cat]}`}>
              <Slider
                min={0}
                max={100}
                value={Math.round(rebalanceConfig.targets[cat] * 100)}
                onChange={v => {
                  const newTargets = { ...rebalanceConfig.targets, [cat]: v / 100 }
                  updateRebalanceConfig({ targets: newTargets })
                }}
                marks={{ 0: '0%', 25: '25%', 50: '50%', 75: '75%', 100: '100%' }}
              />
            </Form.Item>
          ))}
        </Form>
      </Card>

      <Card title="再平衡带宽" style={{ marginBottom: 16 }}>
        <Form layout="vertical">
          <Form.Item label="带宽预设">
            <Select
              value={rebalanceConfig.bandPreset}
              onChange={v => {
                const preset = v as BandPreset
                const bands = BAND_PRESETS[preset]
                updateRebalanceConfig({ bandPreset: preset, bands: { ...bands } })
              }}
              options={BAND_PRESET_OPTIONS}
              style={{ width: 200 }}
            />
          </Form.Item>
          <Space size="large">
            <Form.Item label="下限 (%)">
              <InputNumber
                value={Math.round(rebalanceConfig.bands.low * 100)}
                onChange={v => updateRebalanceConfig({
                  bands: { ...rebalanceConfig.bands, low: (v ?? 15) / 100 },
                  bandPreset: 'CUSTOM',
                })}
                min={0}
                max={50}
                style={{ width: 120 }}
              />
            </Form.Item>
            <Form.Item label="上限 (%)">
              <InputNumber
                value={Math.round(rebalanceConfig.bands.high * 100)}
                onChange={v => updateRebalanceConfig({
                  bands: { ...rebalanceConfig.bands, high: (v ?? 35) / 100 },
                  bandPreset: 'CUSTOM',
                })}
                min={50}
                max={100}
                style={{ width: 120 }}
              />
            </Form.Item>
          </Space>
        </Form>
      </Card>

      <Card title="交易参数" style={{ marginBottom: 16 }}>
        <Form layout="vertical">
          <Space size="large">
            <Form.Item label="最小成交额">
              <InputNumber
                value={rebalanceConfig.minTradeAmount}
                onChange={v => updateRebalanceConfig({ minTradeAmount: v ?? 100 })}
                min={0}
                style={{ width: 160 }}
              />
            </Form.Item>
            <Form.Item label="费用率 (%)">
              <InputNumber
                value={Math.round(rebalanceConfig.feeRate * 10000) / 100}
                onChange={v => updateRebalanceConfig({ feeRate: (v ?? 0.1) / 100 })}
                min={0}
                step={0.01}
                style={{ width: 160 }}
              />
            </Form.Item>
            <Form.Item label="滑点 (%)">
              <InputNumber
                value={Math.round(rebalanceConfig.slippage * 10000) / 100}
                onChange={v => updateRebalanceConfig({ slippage: (v ?? 0.1) / 100 })}
                min={0}
                step={0.01}
                style={{ width: 160 }}
              />
            </Form.Item>
          </Space>
        </Form>
      </Card>

      <Divider />

      <Space>
        <Button onClick={resetToDefaults}>恢复默认设置</Button>
      </Space>

      <Divider />

      <Alert
        message="风险免责声明"
        description="本工具仅提供信息参考，不构成投资建议。不提供交易执行功能。数据可能有延迟，投资决策请咨询专业人士。"
        type="warning"
        showIcon
      />
    </div>
  )
}
```
