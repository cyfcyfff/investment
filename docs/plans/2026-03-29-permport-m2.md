```ts
import { Category, CATEGORIES } from '../types'
import type { AssetHolding, RebalanceConfig, RebalancePlan, RebalanceTrade } from '../types'
import { calculateCategoryWeights, isRebalanceTriggered } from './calcService'
import { generateId } from '../utils/formatters'

function getFxRate(currency: string, baseCurrency: string, fxRates: Record<string, number>): number {
  if (currency === baseCurrency) return 1
  return fxRates[`${currency}-${baseCurrency}`] ?? 1
}

function calcHoldingValue(holding: AssetHolding, prices: Record<string, number>, fxRates: Record<string, number>, baseCurrency: string): number {
  const price = prices[holding.ticker]
  if (price === undefined) return 0
  const fx = getFxRate(holding.currency, baseCurrency, fxRates)
  return holding.quantity * price * fx
}

function calcCategoryValue(
  holdings: AssetHolding[],
  category: Category,
  prices: Record<string, number>,
  fxRates: Record<string, number>,
  baseCurrency: string,
): number {
  return holdings
    .filter(h => h.category === category)
    .reduce((sum, h) => sum + calcHoldingValue(h, prices, fxRates, baseCurrency), 0)
}

export function generateRebalancePlan(
  holdings: AssetHolding[],
  prices: Record<string, number>,
  fxRates: Record<string, number>,
  config: RebalanceConfig,
  baseCurrency: string = 'USD',
): RebalancePlan {
  const currentWeights = calculateCategoryWeights(holdings, prices, fxRates, baseCurrency)

  const totalValue = CATEGORIES.reduce((sum, cat) => {
    return sum + calcCategoryValue(holdings, cat, prices, fxRates, baseCurrency)
  }, 0)

  const check = isRebalanceTriggered(currentWeights, config)
  const triggerReason = check.triggered
    ? (check.breachCategories.length > 0 ? 'BAND_BREACH' as const : 'ANNUAL_REVIEW' as const)
    : 'MANUAL' as const

  if (totalValue === 0 || holdings.length === 0) {
    return {
      id: generateId(),
      generatedAt: new Date().toISOString(),
      triggerReason,
      triggerDetails: check.reason || '手动生成',
      currentWeights,
      targetWeights: config.targets,
      trades: [],
      postWeights: currentWeights,
      totalEstimatedFee: 0,
      totalEstimatedSlippage: 0,
      warnings: [],
    }
  }

  // Calculate delta for each category
  const deltas: Array<{ category: Category; delta: number; absDelta: number }> = []
  for (const cat of CATEGORIES) {
    const targetValue = totalValue * config.targets[cat]
    const currentValue = calcCategoryValue(holdings, cat, prices, fxRates, baseCurrency)
    const delta = targetValue - currentValue
    deltas.push({ category: cat, delta, absDelta: Math.abs(delta) })
  }

  // Sort by absolute delta descending
  deltas.sort((a, b) => b.absDelta - a.absDelta)

  const trades: RebalanceTrade[] = []
  let priority = 1
  const warnings: string[] = []

  // First pass: SELL trades (overweight categories)
  for (const { category, delta } of deltas) {
    if (delta >= 0) continue // Not overweight

    const categoryHoldings = holdings.filter(h => h.category === category)
    const totalCategoryValue = calcCategoryValue(holdings, category, prices, fxRates, baseCurrency)

    for (const holding of categoryHoldings) {
      const holdingValue = calcHoldingValue(holding, prices, fxRates, baseCurrency)
      if (holdingValue === 0 || totalCategoryValue === 0) continue

      // Proportional distribution within category
      const proportion = holdingValue / totalCategoryValue
      const sellAmount = Math.abs(delta) * proportion
      const price = prices[holding.ticker]
      if (price === undefined || price === 0) continue

      const quantity = sellAmount / price
      const fee = sellAmount * config.feeRate
      const slippage = sellAmount * config.slippage

      if (sellAmount < config.minTradeAmount && sellAmount > 0) {
        warnings.push(
          `${holding.ticker} 卖出金额 ${sellAmount.toFixed(2)} 低于最小成交额 ${config.minTradeAmount}`,
        )
      }

      trades.push({
        category,
        holdingId: holding.id,
        ticker: holding.ticker,
        side: 'SELL',
        quantity: Math.round(quantity * 10000) / 10000,
        estimatedPrice: price,
        estimatedAmount: sellAmount,
        estimatedFee: Math.round(fee * 100) / 100,
        priority,
      })
    }
    priority++
  }

  // Calculate total sell proceeds
  const totalSellAmount = trades
    .filter(t => t.side === 'SELL')
    .reduce((sum, t) => sum + t.estimatedAmount, 0)

  // Second pass: BUY trades (underweight categories)
  let buyBudget = totalSellAmount
  for (const { category, delta } of deltas) {
    if (delta <= 0) continue // Not underweight

    const categoryHoldings = holdings.filter(h => h.category === category)

    if (categoryHoldings.length === 0) {
      warnings.push(`${category} 类别无持仓，无法生成买入建议`)
      continue
    }

    for (const holding of categoryHoldings) {
      const holdingValue = calcHoldingValue(holding, prices, fxRates, baseCurrency)
      const totalCategoryValue = calcCategoryValue(holdings, category, prices, fxRates, baseCurrency)
      if (totalCategoryValue === 0) continue

      const proportion = holdingValue / totalCategoryValue
      const buyAmount = delta * proportion

      const price = prices[holding.ticker]
      if (price === undefined || price === 0) continue

      // Cap by available budget
      const actualBuyAmount = Math.min(buyAmount, buyBudget)
      if (actualBuyAmount <= 0) continue

      const quantity = actualBuyAmount / price
      const fee = actualBuyAmount * config.feeRate
      const slippage = actualBuyAmount * config.slippage

      if (actualBuyAmount < config.minTradeAmount) {
        warnings.push(
          `${holding.ticker} 买入金额 ${actualBuyAmount.toFixed(2)} 低于最小成交额 ${config.minTradeAmount}`,
        )
      }

      trades.push({
        category,
        holdingId: holding.id,
        ticker: holding.ticker,
        side: 'BUY',
        quantity: Math.round(quantity * 10000) / 10000,
        estimatedPrice: price,
        estimatedAmount: actualBuyAmount,
        estimatedFee: Math.round(fee * 100) / 100,
        priority,
      })

      buyBudget -= actualBuyAmount + fee + slippage
    }
    priority++
  }

  const totalEstimatedFee = trades.reduce((sum, t) => sum + t.estimatedFee, 0)
  const totalEstimatedSlippage = trades.reduce((sum, t) => {
    return sum + t.estimatedAmount * config.slippage
  }, 0)

  // Calculate post-weights
  const postCategoryValues: Record<Category, number> = {} as Record<Category, number>
  for (const cat of CATEGORIES) {
    let value = calcCategoryValue(holdings, cat, prices, fxRates, baseCurrency)
    for (const trade of trades) {
      if (trade.category !== cat) continue
      const price = prices[trade.ticker]
      if (price === undefined || price === 0) continue
      if (trade.side === 'SELL') {
        value -= trade.quantity * price
      } else {
        value += trade.quantity * price
      }
    }
    postCategoryValues[cat] = value
  }

  const postTotalValue = CATEGORIES.reduce((sum, cat) => sum + postCategoryValues[cat], 0)
  const postWeights: Record<Category, number> = {} as Record<Category, number>
  for (const cat of CATEGORIES) {
    postWeights[cat] = postTotalValue > 0 ? postCategoryValues[cat] / postTotalValue : 0
  }

  return {
    id: generateId(),
    generatedAt: new Date().toISOString(),
    triggerReason,
    triggerDetails: check.reason || '手动生成',
    currentWeights,
    targetWeights: config.targets,
    trades,
    postWeights,
    totalEstimatedFee: Math.round(totalEstimatedFee * 100) / 100,
    totalEstimatedSlippage: Math.round(totalEstimatedSlippage * 100) / 100,
    warnings,
  }
}
```

```ts
import dayjs from 'dayjs'
import utc from 'dayjs/plugin/utc'
import timezone from 'dayjs/plugin/timezone'
import { db } from '../db'
import { calculateCategoryWeights, calcMarketValue } from './calcService'
import { generateId } from '../utils/formatters'
import { Category, CATEGORIES } from '../types'
import type { AssetHolding, PortfolioSnapshot, SnapshotTrigger } from '../types'

dayjs.extend(utc)
dayjs.extend(timezone)

export function getTodayKeyBeijing(): string {
  return dayjs().tz('Asia/Shanghai').format('YYYY-MM-DD')
}

function getFxRate(currency: string, baseCurrency: string, fxRates: Record<string, number>): number {
  if (currency === baseCurrency) return 1
  return fxRates[`${currency}-${baseCurrency}`] ?? 1
}

export async function createSnapshot(
  holdings: AssetHolding[],
  prices: Record<string, number>,
  fxRates: Record<string, number>,
  baseCurrency: string,
  trigger: SnapshotTrigger,
): Promise<PortfolioSnapshot | null> {
  if (holdings.length === 0) return null

  const weights = calculateCategoryWeights(holdings, prices, fxRates, baseCurrency)
  let totalValue = 0

  const holdingEntries = holdings.map(h => {
    const price = prices[h.ticker]
    if (price === undefined) {
      return { holdingId: h.id, marketValue: 0, weight: 0 }
    }
    const fx = getFxRate(h.currency, baseCurrency, fxRates)
    const marketValue = calcMarketValue(h.quantity, price, fx)
    totalValue += marketValue
    return { holdingId: h.id, marketValue, weight: 0 }
  })

  // Recalculate weights based on actual total
  for (const entry of holdingEntries) {
    entry.weight = totalValue > 0 ? entry.marketValue / totalValue : 0
  }

  const snapshot: PortfolioSnapshot = {
    id: generateId(),
    timestamp: new Date().toISOString(),
    trigger,
    totalValue,
    baseCurrency,
    categoryWeights: weights,
    holdings: holdingEntries,
    fxRates: { ...fxRates },
  }

  await db.snapshots.add(snapshot)
  return snapshot
}

export async function ensureDailySnapshot(
  holdings: AssetHolding[],
  prices: Record<string, number>,
  fxRates: Record<string, number>,
  baseCurrency: string,
): Promise<PortfolioSnapshot | null> {
  if (holdings.length === 0) return null

  const todayKey = getTodayKeyBeijing()

  // Check if today's snapshot already exists
  const todayStart = dayjs.tz(todayKey, 'Asia/Shanghai').startOf('day').toISOString()
  const todayEnd = dayjs.tz(todayKey, 'Asia/Shanghai').endOf('day').toISOString()

  const existing = await db.snapshots
    .where('timestamp')
    .between(todayStart, todayEnd, true, true)
    .first()

  if (existing) return existing

  return createSnapshot(holdings, prices, fxRates, baseCurrency, 'DAILY')
}
```

```ts
import type { PortfolioSnapshot, PerformanceMetrics } from '../types'
import { EMPTY_PERFORMANCE } from '../types'

export function calculatePerformance(snapshots: PortfolioSnapshot[]): PerformanceMetrics {
  if (snapshots.length < 2) {
    return { ...EMPTY_PERFORMANCE, snapshotsCount: snapshots.length }
  }

  // Sort by timestamp ascending
  const sorted = [...snapshots].sort(
    (a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime(),
  )

  const first = sorted[0]
  const last = sorted[sorted.length - 1]
  const startValue = first.totalValue
  const endValue = last.totalValue

  // Total return
  const totalReturn = startValue !== 0 ? (endValue - startValue) / startValue : 0
  const totalReturnAmount = endValue - startValue

  // CAGR
  const startMs = new Date(first.timestamp).getTime()
  const endMs = new Date(last.timestamp).getTime()
  const days = (endMs - startMs) / (1000 * 60 * 60 * 24)
  const cagr = startValue > 0 && days > 0
    ? Math.pow(endValue / startValue, 365 / days) - 1
    : 0

  // Max drawdown and peak tracking
  let peakValue = sorted[0].totalValue
  let peakDate = sorted[0].timestamp
  let maxDrawdown = 0
  let maxDrawdownDate = sorted[0].timestamp

  for (const snapshot of sorted) {
    if (snapshot.totalValue > peakValue) {
      peakValue = snapshot.totalValue
      peakDate = snapshot.timestamp
    }

    const drawdown = peakValue > 0 ? 1 - snapshot.totalValue / peakValue : 0
    if (drawdown > maxDrawdown) {
      maxDrawdown = drawdown
      maxDrawdownDate = snapshot.timestamp
    }
  }

  // Current drawdown
  const currentDrawdown = peakValue > 0 ? 1 - endValue / peakValue : 0

  // Days since peak
  const peakMs = new Date(peakDate).getTime()
  const daysSincePeak = Math.floor((endMs - peakMs) / (1000 * 60 * 60 * 24))

  return {
    totalReturn,
    totalReturnAmount,
    cagr,
    maxDrawdown,
    maxDrawdownDate,
    currentDrawdown,
    peakValue,
    peakDate,
    daysSincePeak: Math.max(0, daysSincePeak),
    snapshotsCount: sorted.length,
    firstSnapshotDate: first.timestamp,
    lastSnapshotDate: last.timestamp,
  }
}
```

```ts
  addTransaction: (tx: Omit<Transaction, 'id' | 'createdAt'>) => Promise<void>
  loadTransactions: () => Promise<void>

  transactions: [],
  snapshots: [],
  loading: false,

  loadHoldings: async () => {
    set({ loading: true })
    const holdings = await db.holdings.toArray()
    const transactions = await db.transactions.toArray()
    set({ holdings, transactions, loading: false })
  },

  addHolding: async (data) => {
    const now = new Date().toISOString()
    const normalizedTicker = data.ticker.trim().toUpperCase()
    const holding: AssetHolding = { ...data, ticker: normalizedTicker, id: generateId(), createdAt: now, updatedAt: now }

      date: data.buyDate,
      quantity: data.quantity,

      fee: data.fee,
      totalAmount: data.quantity * data.buyPrice + data.fee,
      notes: `初始买入 ${normalizedTicker}`,
      createdAt: now,
    }
    await db.transactions.add(tx)
    set(state => ({ holdings: [...state.holdings, holding], transactions: [...state.transactions, tx] }))
  },

  updateHolding: async (id, updates) => {
    const updatedAt = new Date().toISOString()
    const normalizedUpdates: Partial<AssetHolding> = updates.ticker != null
      ? { ...updates, ticker: updates.ticker.trim().toUpperCase() }
      : updates
    await db.holdings.update(id, { ...normalizedUpdates, updatedAt })
    set(state => ({ holdings: state.holdings.map(h => h.id === id ? { ...h, ...normalizedUpdates, updatedAt } : h) }))
  },

  deleteHolding: async (id) => {
    await db.holdings.delete(id)
    await db.transactions.where('holdingId').equals(id).delete()
    set(state => ({
      holdings: state.holdings.filter(h => h.id !== id),
      transactions: state.transactions.filter(t => t.holdingId !== id),
    }))
  },

  addTransaction: async (data) => {
    const tx: Transaction = { ...data, id: generateId(), createdAt: new Date().toISOString() }
    await db.transactions.add(tx)
    set(state => ({ transactions: [...state.transactions, tx] }))
  },

  loadTransactions: async () => {
    const transactions = await db.transactions.orderBy('date').reverse().toArray()
    set({ transactions })
  },

  loadSnapshots: async () => {
    const snapshots = await db.snapshots.orderBy('timestamp').reverse().toArray()
    set({ snapshots })
  },

  addSnapshot: async (snapshot) => {
    await db.snapshots.add(snapshot)
    set(state => ({ snapshots: [snapshot, ...state.snapshots] }))
  },
}))
```

```tsx
  const { appConfig, rebalanceConfig } = useConfigStore()

  const [plan, setPlan] = useState<RebalancePlan | null>(null)
  const [drawerOpen, setDrawerOpen] = useState(false)
  const [generating, setGenerating] = useState(false)

  useEffect(() => {
    loadHoldings()
  }, [loadHoldings])

  useEffect(() => {
    if (holdings.length > 0) {
      const tickers = holdings.map(h => h.ticker)
      const currencies = holdings.map(h => h.currency)
      refreshAll(tickers, currencies, appConfig.baseCurrency, appConfig.apiKeys.fmp ?? '')
    }
  }, [holdings, appConfig.baseCurrency, appConfig.apiKeys, refreshAll])

  const prices = useMemo(() => {
    const p: Record<string, number> = {}
    for (const [ticker, quote] of Object.entries(quotes)) {
      p[ticker] = quote.price
    }
    return p
  }, [quotes])

  const fxMap = useMemo(() => {
    const m: Record<string, number> = {}
    for (const [key, fx] of Object.entries(fxRates)) {
      m[key] = fx.rate
    }
    return m
  }, [fxRates])

  const weights = useMemo(
    () => calculateCategoryWeights(holdings, prices, fxMap, appConfig.baseCurrency),
    [holdings, prices, fxMap, appConfig.baseCurrency],
  )

  const rebalanceCheck = useMemo(
    () => isRebalanceTriggered(weights, rebalanceConfig),
    [weights, rebalanceConfig],
  )

  const handleGenerate = () => {
    setGenerating(true)
    try {
      const newPlan = generateRebalancePlan(holdings, prices, fxMap, rebalanceConfig, appConfig.baseCurrency)
      setPlan(newPlan)
      if (newPlan.trades.length === 0) {
        message.info('当前无需再平衡操作')
      }
    } catch (e) {
      message.error('生成建议失败: ' + String(e))
    } finally {
      setGenerating(false)
    }
  }

  const handleRecordTrade = async (data: TradeRecordFormData) => {
    if (!plan) return
    const trade = plan.trades.find(t => t.holdingId === data.holdingId)
    if (!trade) return

    await addTransaction({
      holdingId: data.holdingId,
      type: trade.side === 'SELL' ? 'REBALANCE_OUT' : 'REBALANCE_IN',
      date: data.date,
      quantity: data.actualQuantity,
      price: data.actualPrice,
      fee: data.actualFee,
      totalAmount: data.actualQuantity * data.actualPrice + data.actualFee,
      notes: data.notes || `再平衡: ${trade.side === 'SELL' ? '卖出' : '买入'} ${trade.ticker}`,
    })

    // Create snapshot after recording
    await createSnapshot(holdings, prices, fxMap, appConfig.baseCurrency, 'REBALANCE')

    setDrawerOpen(false)
    message.success('交易记录已保存')
  }

  const columns = [
    {
      title: '优先级',
      dataIndex: 'priority',
      key: 'priority',
      width: 80,
    },
    {
      title: '标的',
      dataIndex: 'ticker',
      key: 'ticker',
    },
    {
      title: '类别',
      dataIndex: 'category',
      key: 'category',
      render: (cat: Category) => <Tag>{CATEGORY_LABELS[cat]}</Tag>,
    },
    {
      title: '方向',
      dataIndex: 'side',
      key: 'side',
      render: (side: string) => (
        <Tag color={side === 'SELL' ? 'red' : 'green'}>{side === 'SELL' ? '卖出' : '买入'}</Tag>
      ),
    },
    {
      title: '数量',
      dataIndex: 'quantity',
      key: 'quantity',
      render: (q: number) => q.toFixed(4),
    },
    {
      title: '预估价格',
      dataIndex: 'estimatedPrice',
      key: 'estimatedPrice',
      render: (p: number) => formatCurrency(p, appConfig.baseCurrency),
    },
    {
      title: '预估金额',
      dataIndex: 'estimatedAmount',
      key: 'estimatedAmount',
      render: (a: number) => formatCurrency(a, appConfig.baseCurrency),
    },
    {
      title: '预估费用',
      dataIndex: 'estimatedFee',
      key: 'estimatedFee',
      render: (f: number) => formatCurrency(f, appConfig.baseCurrency),
    },
  ]

  if (holdings.length === 0) {
    return (
      <div>
        <Title level={3}>再平衡</Title>
        <Empty description="暂无持仓，请先前往「持仓管理」添加">
          <Button type="primary">前往添加</Button>
        </Empty>
      </div>
    )
  }

  return (
    <div>
      <Title level={3}>再平衡</Title>

      {/* Status */}
      <Card style={{ marginBottom: 16 }}>
        <Space>
          {rebalanceCheck.triggered ? (
            <Alert
              message="需要再平衡"
              description={rebalanceCheck.reason}
              type="warning"
              icon={<WarningOutlined />}
              showIcon
            />
          ) : (
            <Alert
              message="无需再平衡"
              description="所有资产类别权重在带宽范围内"
              type="success"
              icon={<CheckCircleOutlined />}
              showIcon
            />
          )}
        </Space>
      </Card>

      {/* Generate button */}
      <Button
        type="primary"
        icon={<ThunderboltOutlined />}
        onClick={handleGenerate}
        loading={generating}
        size="large"
        style={{ marginBottom: 24 }}
      >
        生成再平衡建议
      </Button>

      {/* Plan results */}
      {plan && (
        <>
          {plan.trades.length === 0 ? (
            <Alert
              message="当前无需再平衡操作"
              description="所有资产类别权重在目标范围内"
              type="info"
              showIcon
              style={{ marginBottom: 24 }}
            />
          ) : (
            <>
              {/* Trades table */}
              <Card title="交易建议列表" style={{ marginBottom: 16 }}>
                <Table
                  dataSource={plan.trades}
                  columns={columns}
                  rowKey={(r: RebalanceTrade) => `${r.holdingId}-${r.side}`}
                  pagination={false}
                  size="middle"
                />
              </Card>

              {/* Weight comparison */}
              <Card title="权重对比" style={{ marginBottom: 16 }}>
                {loading ? <Spin /> : (
                  <WeightComparisonChart
                    currentWeights={plan.currentWeights}
                    targetWeights={plan.targetWeights}
                    postWeights={plan.postWeights}
                    config={rebalanceConfig}
                  />
                )}
              </Card>

              {/* Summary */}
              <Card title="汇总" style={{ marginBottom: 16 }}>
                <Space size="large">
                  <Statistic
                    title="预估总费用"
                    value={plan.totalEstimatedFee}
                    formatter={v => formatCurrency(Number(v), appConfig.baseCurrency)}
                  />
                  <Statistic
                    title="预估滑点"
                    value={plan.totalEstimatedSlippage}
                    formatter={v => formatCurrency(Number(v), appConfig.baseCurrency)}
                  />
                </Space>
                {plan.warnings.length > 0 && (
                  <div style={{ marginTop: 16 }}>
                    {plan.warnings.map((w, i) => (
                      <Alert key={i} message={w} type="warning" showIcon icon={<ExclamationCircleOutlined />} style={{ marginBottom: 8 }} />
                    ))}
                  </div>
                )}
              </Card>

              <Divider />

              <Alert
                message="风险免责声明"
                description="本建议仅供参考，不构成投资建议。不提供交易执行功能。请根据实际情况调整后在外部券商执行，然后点击下方按钮记录实际交易。"
                type="warning"
                showIcon
                style={{ marginBottom: 16 }}
              />

              <Button type="primary" onClick={() => setDrawerOpen(true)}>
                记录执行结果
              </Button>
            </>
          )}
        </>
      )}

      {/* Trade record drawer */}
      <Drawer
        title="记录再平衡交易"
        width={500}
        open={drawerOpen}
        onClose={() => setDrawerOpen(false)}
        destroyOnClose
      >
        {plan && plan.trades.length > 0 && (
          <TradeRecordForm
            trades={plan.trades}
            onSubmit={handleRecordTrade}
            onCancel={() => setDrawerOpen(false)}
          />
        )}
      </Drawer>
    </div>
  )
}
```

```tsx
import { AreaChart, Area, XAxis, YAxis, CartesianGrid, Tooltip, ResponsiveContainer, ReferenceLine } from 'recharts'
import type { PortfolioSnapshot } from '../../types'

interface NetValueChartProps {
  snapshots: PortfolioSnapshot[]
  baseCurrency: string
}

export default function NetValueChart({ snapshots, baseCurrency }: NetValueChartProps) {
  if (snapshots.length < 2) {
    return <div style={{ textAlign: 'center', padding: 40, color: '#999' }}>需要至少 2 个快照数据</div>
  }

  const sorted = [...snapshots].sort(
    (a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime(),
  )

  const data = sorted.map(s => ({
    date: new Date(s.timestamp).toLocaleDateString('zh-CN', { month: '2-digit', day: '2-digit' }),
    fullDate: new Date(s.timestamp).toLocaleDateString('zh-CN'),
    总市值: Math.round(s.totalValue * 100) / 100,
  }))

  const values = data.map(d => d['总市值'])
  const minVal = Math.min(...values)
  const maxVal = Math.max(...values)

  return (
    <ResponsiveContainer width="100%" height={300}>
      <AreaChart data={data} margin={{ top: 10, right: 30, left: 20, bottom: 0 }}>
        <defs>
          <linearGradient id="netValueGradient" x1="0" y1="0" x2="0" y2="1">
            <stop offset="5%" stopColor="#1890ff" stopOpacity={0.3} />
            <stop offset="95%" stopColor="#1890ff" stopOpacity={0} />
          </linearGradient>
        </defs>
        <CartesianGrid strokeDasharray="3 3" />
        <XAxis dataKey="date" tick={{ fontSize: 12 }} />
        <YAxis
          domain={[Math.floor(minVal * 0.98), Math.ceil(maxVal * 1.02)]}
          tick={{ fontSize: 12 }}
          tickFormatter={(v: number) => `${(v / 1000).toFixed(0)}k`}
        />
        <Tooltip
          formatter={(value: number) => [`${value.toLocaleString()} ${baseCurrency}`, '总市值']}
          labelFormatter={(label: string) => {
            const entry = data.find(d => d.date === label)
            return entry?.fullDate ?? label
          }}
        />
        <ReferenceLine
          y={data[0]['总市值']}
          stroke="#999"
          strokeDasharray="5 5"
          label={{ value: '起始', position: 'right', fontSize: 11 }}
        />
        <Area
          type="monotone"
          dataKey="总市值"
          stroke="#1890ff"
          strokeWidth={2}
          fill="url(#netValueGradient)"
        />
      </AreaChart>
    </ResponsiveContainer>
  )
}
```
